// Secure Firestore Security Rules for WhatsApp Sentinel
// These rules ensure only authenticated users can access their own data

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // WhatsApp collection with user subcollections - only authenticated users can access their own data
    match /whatsapp/{userId} {
      // Allow read and write only if user is authenticated and accessing their own data
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Allow create only if user is authenticated and creating their own document
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Conversations subcollection - only the user can access their own conversations
      match /conversations/{conversationId} {
        // Allow read and write only if user is authenticated and accessing their own conversations
        allow read, write: if request.auth != null && request.auth.uid == userId;
        
        // Allow create only if user is authenticated and creating their own conversation
        allow create: if request.auth != null && request.auth.uid == userId;
      }
    }
    
    // Alternative: More restrictive rules for production
    // Uncomment these rules for stricter security
    
    /*
    match /whatsapp/{userId} {

      // Conversations subcollection with stricter rules
      match /conversations/{conversationId} {
        // Only allow authenticated users to read their own conversations with retention policy
        allow read: if request.auth != null 
          && request.auth.uid == userId
          && request.time < resource.data.createdAt + duration.value(30, 'd'); // 30-day retention
        
        // Only allow authenticated users to create their own conversations

        // No updates or deletes allowed (immutable data)

      }
    }
    */
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
