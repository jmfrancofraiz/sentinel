// Secure Firestore Security Rules for WhatsApp Sentinel
// These rules ensure only authenticated users can access their own data

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // WhatsApp collection - only authenticated users can access data
    match /whatsapp/{document} {
      // Allow read and write only if user is authenticated
      allow read, write: if request.auth != null;
      
      // Allow create only if user is authenticated
      allow create: if request.auth != null;
    }
    
    // Alternative: More restrictive rules for production
    // Uncomment these rules for stricter security
    
    /*
    match /whatsapp/{document} {
      // Only allow authenticated users to read data
      allow read: if request.auth != null 
        && request.time < resource.data.createdAt + duration.value(30, 'd'); // 30-day retention
      
      // Only allow authenticated users to create documents
      allow create: if request.auth != null 
        && request.resource.data.keys().hasAll(['timestamp', 'createdAt'])
        && request.resource.data.createdAt is number;
      
      // No updates or deletes allowed (immutable data)
      allow update, delete: if false;
    }
    */
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
