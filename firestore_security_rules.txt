// Secure Firestore Security Rules for WhatsApp Sentinel
// These rules ensure only authenticated users can access their own data

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // WhatsApp collection with user subcollections - only authenticated users can access their own data
    match /whatsapp/{userId} {
      // Allow read and write only if user is authenticated and accessing their own data
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Allow create only if user is authenticated and creating their own document
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Conversations collection - only the user can access their own conversations
      match /conversations {
        // Allow read and write only if user is authenticated and accessing their own data
        allow read, write: if request.auth != null && request.auth.uid == userId;
        
        // Participant/Group documents - only the user can access their own conversations
        match /{participantOrGroup} {
          // Allow read and write only if user is authenticated and accessing their own data
          allow read, write: if request.auth != null && request.auth.uid == userId;
          
          // Interactions subcollection - only the user can access their own interactions
          match /interactions/{interactionId} {
            // Allow read and write only if user is authenticated and accessing their own interactions
            allow read, write: if request.auth != null && request.auth.uid == userId;
            
            // Allow create only if user is authenticated and creating their own interaction
            allow create: if request.auth != null && request.auth.uid == userId;
          }
        }
      }
    }
